// Tu clave API de OMDb
const apiKey = '4526760c';

// URL base de la API
const urlBase = 'https://www.omdbapi.com/';

// Función para traer las películas destacadas
function obtenerPeliculasDestacadas() {
  // Aquí puedes cambiar el término de búsqueda según lo que necesites
  const parametros = {
    s: 'movie', // término de búsqueda
    type: 'movie', // solo películas
    y: 2025, // año, opcional
    page: 1, // página de resultados
    apikey: apiKey // tu clave API
  };

  const url = new URL(urlBase);
  Object.keys(parametros).forEach(key => url.searchParams.append(key, parametros[key]));

  fetch(url)
    .then(response => {
      if (!response.ok) {
        throw new Error('Error en la solicitud a OMDb API');
      }
      return response.json();
    })
    .then(data => {
      if (data.Response === 'True') {
        mostrarPeliculas(data.Search); // Muestra las películas
      } else {
        console.log('No se encontraron películas');
      }
    })
    .catch(error => {
      console.error('Hubo un problema con la solicitud:', error);
    });
}

// Función para mostrar las películas en la consola o en el DOM
function mostrarPeliculas(peliculas) {
  // Por ejemplo, simplemente mostrar las películas en la consola
  peliculas.forEach(pelicula => {
    console.log(`${pelicula.Title} (${pelicula.Year})`);
    console.log(`Poster: ${pelicula.Poster}`);
    console.log('--------------------------------');
  });

  // O si quieres mostrar en el DOM:
  const listaPeliculas = document.getElementById('lista-peliculas');
  listaPeliculas.innerHTML = ''; // Limpiar la lista
  peliculas.forEach(pelicula => {
    const divPelicula = document.createElement('div');
    divPelicula.classList.add('pelicula');
    divPelicula.innerHTML = `
      <h3>${pelicula.Title} (${pelicula.Year})</h3>
      <img src="${pelicula.Poster}" alt="${pelicula.Title}" />
    `;
    listaPeliculas.appendChild(divPelicula);
  });
}

// Llamar a la función para obtener las películas
obtenerPeliculasDestacadas();
